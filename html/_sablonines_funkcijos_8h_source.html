35template <typename StudentuKonteineris>
36StudentuKonteineris nuskaitytiStudentuDuomenisIsFailo(const std::string& failas) {
37 StudentuKonteineris studentuSarasas;
38 FILE* skaitomasFailas = std::fopen(failas.c_str(), "r");
39 if (!skaitomasFailas) throw std::runtime_error("Nepavyko atidaryti failo: " + failas);
40 try {
41 static char ivestiesBuferis[1 << 20];
42 std::setvbuf(skaitomasFailas, ivestiesBuferis, _IOFBF, sizeof(ivestiesBuferis));
43 char antraste[257];
44 if (!std::fgets(antraste, sizeof(antraste), skaitomasFailas)) {
45 std::fclose(skaitomasFailas);
46 return studentuSarasas;
47 }
48 std::size_t namuDarbuKiekis = 0;
49 char laikinaEilute[257];
50 std::strncpy(laikinaEilute, antraste, sizeof(laikinaEilute));
51 laikinaEilute[sizeof(laikinaEilute) - 1] = '\0';
52 char* stulpelis = std::strtok(laikinaEilute, " \t\r\n");
53 while (stulpelis) {
54 if (stulpelis[0] == 'N' && stulpelis[1] == 'D') namuDarbuKiekis++;
55 stulpelis = std::strtok(nullptr, " \t\r\n");
56 }
57 char eilute[1024];
58 while (std::fgets(eilute, sizeof(eilute), skaitomasFailas)) {
59 try {
60 Studentas studentas(eilute, namuDarbuKiekis);
61 studentuSarasas.push_back(std::move(studentas));
62 } catch (...) {
63 continue;
64 }
65 }
66 std::fclose(skaitomasFailas);
67 return studentuSarasas;
68 }
69 catch (const std::bad_alloc&) {
70 std::fclose(skaitomasFailas);
71 std::cerr << "Nepakanka atminties...\n";
72 return studentuSarasas;
73 }
74}
