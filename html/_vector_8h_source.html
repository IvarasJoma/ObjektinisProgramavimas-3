553 iterator insert(const_iterator cpos, size_type n, const value_type& val) {
554 if (n == 0) return const_cast<iterator>(cpos);
555 size_type index = static_cast<size_type>(cpos - dat);
556
557 Vector tmp;
558 tmp.reallocate(size() + n);
559 std::uninitialized_copy(dat, dat + index, tmp.dat);
560 std::uninitialized_fill(tmp.dat + index, tmp.dat + index + n, val);

562 tmp.avail = tmp.dat + size() + n;

825 void shift_right(iterator pos, size_type n) {
826 if (pos == avail) return;
827 size_type tail = static_cast<size_type>(avail - pos);
828 if (n >= tail) {
829 // Visi elementai [pos, avail) patenka į neinicializuotą sritį
830 std::uninitialized_move(pos, avail, pos + n);
831 } else {
832 // Paskutiniai n elementų patenka į neinicializuotą sritį
833 std::uninitialized_move(avail - n, avail, avail);
834 // Likę elementai stumiami inicializuotoje srityje
835 std::move_backward(pos, avail - n, avail);
836 }
