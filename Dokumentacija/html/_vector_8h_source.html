724 template <std::ranges::input_range R>
725 iterator insert_range(const_iterator cpos, R&& r) {
726 size_type index = static_cast<size_type>(cpos - dat);
727
728 if constexpr (std::ranges::sized_range<R>) {
729 size_type n = static_cast<size_type>(std::ranges::distance(r));
730 if (n == 0) return dat + index;
731
732 if (size() + n <= capacity()) {
733 iterator pos = dat + index;
734 shift_right(pos, n);
735 iterator out = pos;
736 for (auto&& elem : r)
737 alloc_traits::construct(alloc, out++,
738 std::forward<decltype(elem)>(elem));
739 avail += n;
740 return pos;
741 } else {
742 // reallocation path (same as before)
743 Vector tmp(alloc);
744 tmp.reallocate(grow_capacity(size() + n));
745 std::uninitialized_move(dat, dat + index, tmp.dat);
746 std::uninitialized_copy(std::ranges::begin(r),
747 std::ranges::end(r),
748 tmp.dat + index);
749 std::uninitialized_move(dat + index, avail,
750 tmp.dat + index + n);
751 tmp.avail = tmp.dat + size() + n;
752 swap(tmp);
753 return dat + index;
754 }
755 } else {
756 // Non-sized / single-pass: collect into a temporary Vector first,
757 // then delegate to the sized overload
758 Vector<T, Allocator> tmp_range(alloc);
759 for (auto&& elem : r)
760 tmp_range.push_back(std::forward<decltype(elem)>(elem));

1043 void shift_right(iterator pos, size_type n) {
1044 if (pos == avail) return;
1045
1046 if constexpr (std::is_trivially_copyable_v<T>) {
1047 std::memmove(pos + n, pos,
1048 static_cast<std::size_t>(avail - pos) * sizeof(T));
1049 return;
1050 }
1051 size_type tail = static_cast<size_type>(avail - pos);
1052 if (n >= tail) {
1053 iterator src = pos, dst = pos + n;
1054 for (; src != avail; ++src, ++dst)
1055 alloc_traits::construct(alloc, dst, std::move_if_noexcept(*src));
1056 destroy_range(pos, avail);
1057 } else {
1058 iterator uninit_src = avail - n;
1059 iterator uninit_dst = avail;
1060 for (iterator it = uninit_src; it != avail; ++it, ++uninit_dst)
1061 alloc_traits::construct(alloc, uninit_dst,
1062 std::move_if_noexcept(*it));
1063 for (iterator it = uninit_src; it != pos; ) {
1064 --it;
1065 *(it + n) = std::move_if_noexcept(*it);
1066 }
1067 }
